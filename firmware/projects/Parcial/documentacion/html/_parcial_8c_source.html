81static void taskMedir(void *pvParameter)
82{
83 uint16_t dato;
84 while (true)
85 {
86 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
87 if (sistema)
88 {
89 AnalogInputReadSingle(CH1, &dato);
90 pH = (float)dato * (14.0/3.0) ; // como el voltaje es de 0 a 3, y 0V es 0 pH, y 3V es 14 pH, el factor de conversion es 14/3
91 if(pH < 6.0)
92 {
93 //prendo bobmba basica y apago la acida
94 GPIOOn(GPIO_23);
95 GPIOOff(GPIO_22);
96 bomba_acido = false;
97 bomba_base = true;
98 }
99 else
100 if(pH > 6.7)
101 {
102 //prondo bomba acida y apago la basica
103 GPIOOn(GPIO_22);
104 GPIOOff(GPIO_23);
105 bomba_base = false;
106 bomba_acido = true;
107 }
108 else
109 {
110 // apago las dos bombas
111 GPIOOff(GPIO_22);
112 GPIOOff(GPIO_23);
113 bomba_acido = false;
114 bomba_base = false;
115 }
116
117 // para la humedad si da 1 hay que prender la bomba si da 0 hay que apagarla
118 bomba_agua = GPIORead(GPIO_20);
119 if (bomba_agua)

121 GPIOOn(GPIO_20);
122 }

124 {
125 GPIOOff(GPIO_20);
