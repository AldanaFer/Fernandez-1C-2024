83static void TaskSenial(void *pvParameter)
84{
85 float emg[CHUNK] = {0}; // Arreglo que se usa para guardar de a 4 lecturas de emg
86 float emg_filtrado[CHUNK] = {0}; // Arreglo que se usa para guardar de a 4 valores de emg filtrado
87 uint16_t contador_emg_completo = 0;
88 uint16_t lectura = 0;
89 char msg[128];
90 char msg_chunk[24];
91 char msg_filtrado[128];
92 char msg_chunk_filtrado[24];
93 uint16_t indice = 0;
94 while (true)
95 {
96 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
97 AnalogInputReadSingle(CH1, &lectura);
98 emg[indice] = (float)lectura;
99 indice++;
100 if (indice == CHUNK)
101 {
102 HiPassFilter(&emg[0], emg_filtrado, CHUNK);
103 LowPassFilter(emg_filtrado, emg_filtrado, CHUNK);
104 strcpy(msg, "");
105 strcpy(msg_filtrado, "");
106 for (uint8_t i = 0; i < CHUNK; i++)
107 {
108 emg_valor_absoluto[i + contador_emg_completo] = fabs(emg_filtrado[i]);
109 sprintf(msg_chunk, "*A%.2f*", emg[i]);
110 strcat(msg, msg_chunk);
111 sprintf(msg_chunk_filtrado, "*G%.2f*", emg_filtrado[i]);
112 strcat(msg_filtrado, msg_chunk_filtrado);
113 }
114 BleSendString(msg_filtrado);
115 BleSendString(msg);
116 indice = 0;
117 contador_emg_completo = contador_emg_completo + CHUNK;
118 }
119 if (contador_emg_completo >= SIZE_EMG_COMPLETO)
120 {
121 contador_emg_completo = 0;
122 }
123 }
124}

129static void TaskMover(void *pvParameter)
130{
131 bool abrir = false; // Indica si se abrio o cerro el servo
132 float maximo = 0;
133 uint8_t angulo = 90;
134 float umbral = 0.05;
135 while (true)
136 {
137 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
138 maximo = 0;
139 for (int j = 0; j < SIZE_EMG_COMPLETO; j++)
140 {
141 if (maximo < emg_valor_absoluto[j])
142 {
143 maximo = emg_valor_absoluto[j];
144 }
145 }
146 if (maximo >= umbral)
147 {
148 if (abrir)
149 {
150 ServoMove(SERVO_0, angulo);
151 printf("abrir mano\n");
152 abrir = false;
153 }
154 else
155 {
156 ServoMove(SERVO_0, angulo*(-1) );
157 printf("cerrar mano\n");
158 abrir = true;
159 }
160 }
161 }

166void app_main(void)

180 .period = CONFIG_PERIOD_SENIAL_US,
181 .func_p = FuncTimerSenial,
182 .param_p = NULL};

184
185 timer_config_t timer_mover = {

188 .func_p = FuncTimerMover,
189 .param_p = NULL};
190 TimerInit(&timer_mover);
191

193 HiPassInit(SAMPLE_FREQ, FRECUENCIA_PASA_ALTO, ORDER_2);
194
195 xTaskCreate(&TaskSenial, "Senial", 4096, NULL, 5, &senial_task_handle);
196 TimerStart(timer_senial.timer);
197
198 xTaskCreate(&TaskMover, "Mover", 4096, NULL, 5, &mover_task_handle);
199 TimerStart(timer_mover.timer);
200
201 uint8_t blink = 0;
202 static neopixel_color_t color;
203 ble_config_t ble_configuration = {
204 "ESP_ALDU",
205 NULL};
206
207 NeoPixelInit(BUILT_IN_RGB_LED_PIN, BUILT_IN_RGB_LED_LENGTH, &color);
208 NeoPixelAllOff();
209 BleInit(&ble_configuration);
210 while (1)
211 {
212 vTaskDelay(CONFIG_BLINK_PERIOD / portTICK_PERIOD_MS);
213 switch (BleStatus())
214 {
215 case BLE_OFF:
216 NeoPixelAllOff();
217 break;
218 case BLE_DISCONNECTED:
219 if (blink % 2)
220 {
221 NeoPixelAllColor(NEOPIXEL_COLOR_TURQUOISE);
222 }
223 else
224 {
225 NeoPixelAllOff();
226 }
227 blink++;
228 break;
229 case BLE_CONNECTED:
230 NeoPixelAllColor(NEOPIXEL_COLOR_ROSE);
231 break;
232 }
233 }
234};
